## 🔧 API对接

### 1. 启动教学(工具增强)

**端点**:
```http
POST /api/v1/teaching-v2/session/{session_id}/teach-v2?use_tools=true
```

**参数**:
- `session_id`: 学习会话ID
- `use_tools`: 是否启用工具增强 (默认true)

**响应**: SSE事件流

**前端示例**:
```typescript
// hooks/useTeachingV2.ts
import { useEffect, useState } from 'react';

interface TeachingEvent {
  type: 'segment' | 'tool_call' | 'tool_result' | 'complete' | 'error';
  message?: string;
  image_id?: string;
  video_id?: string;
  demo_id?: string;
  whiteboard?: {
    title?: string;
    points?: string[];
    formulas?: string[];
  };
  next_action?: string;
}

export function useTeachingV2(sessionId: string, useTools: boolean = true) {
  const [events, setEvents] = useState<TeachingEvent[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const startTeaching = async () => {
    setIsStreaming(true);
    setError(null);
    setEvents([]);

    const eventSource = new EventSource(
      `/api/v1/teaching-v2/session/${sessionId}/teach-v2?use_tools=${useTools}`
    );

    eventSource.onmessage = (event) => {
      try {
        const data: TeachingEvent = JSON.parse(event.data);
        setEvents(prev => [...prev, data]);

        // Handle completion
        if (data.type === 'complete') {
          eventSource.close();
          setIsStreaming(false);
        }
      } catch (err) {
        console.error('Failed to parse event:', err);
      }
    };

    eventSource.onerror = (err) => {
      console.error('SSE error:', err);
      setError('Connection error');
      eventSource.close();
      setIsStreaming(false);
    };
  };

  return {
    events,
    isStreaming,
    error,
    startTeaching,
  };
}
```

## 🎨 UI组件实现

### 1. 图片渲染组件

```tsx
// components/TeachingImage.tsx
import React, { useEffect, useState } from 'react';
import { getImage } from '@/services/imageService';

interface TeachingImageProps {
  imageId: string;
  alt?: string;
  className?: string;
}

export function TeachingImage({ imageId, alt, className }: TeachingImageProps) {
  const [imageUrl, setImageUrl] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    setLoading(true);
    setError(null);

    getImage(imageId)
      .then(data => {
        setImageUrl(data.url);
        setLoading(false);
      })
      .catch(err => {
        setError(err.message);
        setLoading(false);
      });
  }, [imageId]);

  if (loading) {
    return (
      <div className={`animate-pulse bg-gray-200 rounded-lg ${className}`}>
        <div className="h-48 flex items-center justify-center">
          <span className="text-gray-400">Loading image...</span>
        </div>
      </div>
    );
  }

  if (error) {
    return (
      <div className={`bg-red-50 border border-red-200 rounded-lg p-4 ${className}`}>
        <p className="text-red-600 text-sm">Failed to load image: {error}</p>
      </div>
    );
  }

  return (
    <img
      src={imageUrl!}
      alt={alt || 'Teaching image'}
      className={`rounded-lg shadow-md ${className}`}
    />
  );
}
```

### 2. 教学事件渲染器

```tsx
// components/TeachingEventRenderer.tsx
import React from 'react';
import { TeachingImage } from './TeachingImage';

interface TeachingEvent {
  type: 'segment' | 'tool_call' | 'tool_result' | 'complete' | 'error';
  message?: string;
  image_id?: string;
  video_id?: string;
  demo_id?: string;
  whiteboard?: {
    title?: string;
    points?: string[];
    formulas?: string[];
  };
}

interface TeachingEventRendererProps {
  event: TeachingEvent;
}

export function TeachingEventRenderer({ event }: TeachingEventRendererProps) {
  switch (event.type) {
    case 'segment':
      return (
        <div className="mb-4">
          {/* Message */}
          {event.message && (
            <p className="text-gray-800 mb-2">{event.message}</p>
          )}
          
          {/* Whiteboard content */}
          {event.whiteboard && (
            <div className="bg-gray-50 rounded-lg p-4 mb-2">
              {event.whiteboard.title && (
                <h3 className="font-bold text-lg mb-2">{event.whiteboard.title}</h3>
              )}
              {event.whiteboard.points && (
                <ul className="list-disc list-inside mb-2">
                  {event.whiteboard.points.map((point, i) => (
                    <li key={i} className="text-gray-700">{point}</li>
                  ))}
                </ul>
              )}
              {event.whiteboard.formulas && (
                <div className="space-y-1">
                  {event.whiteboard.formulas.map((formula, i) => (
                    <div key={i} className="font-mono text-blue-600">{formula}</div>
                  ))}
                </div>
              )}
            </div>
          )}
          
          {/* Image */}
          {event.image_id && (
            <TeachingImage imageId={event.image_id} className="w-full max-w-2xl" />
          )}
        </div>
      );

    case 'tool_call':
      return (
        <div className="bg-blue-50 border-l-4 border-blue-500 p-4 mb-4">
          <p className="text-blue-700 text-sm font-medium">
            🔧 Tool call: {event.message}
          </p>
        </div>
      );

    case 'tool_result':
      return (
        <div className="bg-green-50 border-l-4 border-green-500 p-4 mb-4">
          <p className="text-green-700 text-sm font-medium">
            ✓ Tool result: {event.message}
          </p>
          {event.image_id && (
            <TeachingImage imageId={event.image_id} className="w-full max-w-md mt-2" />
          )}
        </div>
      );

    case 'error':
      return (
        <div className="bg-red-50 border-l-4 border-red-500 p-4 mb-4">
          <p className="text-red-700 text-sm font-medium">
            ✗ Error: {event.message}
          </p>
        </div>
      );

    case 'complete':
      return (
        <div className="bg-gray-100 rounded-lg p-4 mt-4">
          <p className="text-gray-600 text-center">✓ Teaching completed</p>
        </div>
      );

    default:
      return null;
  }
}
```

// components/TeachingView.tsx
import React, { useState } from 'react';

interface TeachingViewProps {
  sessionId: string;

export function TeachingView({ sessionId }: TeachingViewProps) {
  const [useTools, setUseTools] = useState(true);
  const { events, isStreaming, error, startTeaching } = useTeachingV2(sessionId, useTools);

  return (
    <div className="max-w-4xl mx-auto p-6">

      <div className="mb-6">

              type="checkbox"
              checked={useTools}
              onChange={(e) => setUseTools(e.target.checked)}
              className="rounded"
            />
            <span className="text-sm text-gray-600">Enable tool enhancement</span>
          </label>
          
          <button
            onClick={startTeaching}
            disabled={isStreaming}
            className="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700 disabled:bg-gray-400"
          >
            {isStreaming ? 'Teaching...' : 'Start Teaching'}
          </button>
        </div>
      </div>

      {/* Error */}
      {error && (
        <div className="bg-red-50 border border-red-200 rounded-lg p-4 mb-4">
          <p className="text-red-600">{error}</p>
        </div>
      )}

      {/* Events */}
      <div className="space-y-4">
        {events.map((event, index) => (
          <TeachingEventRenderer key={index} event={event} />
        ))}
      </div>

      {/* Loading indicator */}
      {isStreaming && events.length === 0 && (
        <div className="text-center py-8">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600 mx-auto"></div>
          <p className="text-gray-600 mt-4">Starting teaching...</p>

## 🔄 SSE事件处理

### 事件类型

| 事件类型 | 说明 | 数据字段 |
|---------|------|---------|
| `segment` | 教学内容片段 | message, whiteboard, image_id |
| `tool_call` | 工具调用 | message, tool_name |
| `tool_result` | 工具结果 | message, image_id, video_id |
| `complete` | 教学完成 | next_action |
| `error` | 错误 | message |

### 错误处理

```typescript
// utils/sseErrorHandler.ts
export function handleSSEError(error: any): string {
  if (error.name === 'AbortError') {
    return 'Connection timeout';
  }
  
  if (error.message?.includes('401')) {
    return 'Authentication required';
  }
  
  if (error.message?.includes('404')) {
    return 'Session not found';
  }
  
  return 'Connection error, please retry';
}
```

### 重试机制

```typescript
// hooks/useTeachingV2WithRetry.ts
import { useState, useCallback } from 'react';
import { useTeachingV2 } from './useTeachingV2';

export function useTeachingV2WithRetry(sessionId: string, maxRetries: number = 3) {
  const [retryCount, setRetryCount] = useState(0);
  const { events, isStreaming, error, startTeaching } = useTeachingV2(sessionId);

  const startTeachingWithRetry = useCallback(async () => {
    try {
      await startTeaching();
      setRetryCount(0);
    } catch (err) {
      if (retryCount < maxRetries) {
        setRetryCount(prev => prev + 1);
        setTimeout(() => startTeaching(), 1000 * (retryCount + 1));
      }
    }
  }, [startTeaching, retryCount, maxRetries]);

  return {
    events,
    isStreaming,
    error,
    retryCount,
    startTeaching: startTeachingWithRetry,
  };
}
```

## 🐛 常见问题

### Q1: SSE连接断开

**原因**: 网络不稳定或超时

**解决**: 实现重连机制

```typescript
eventSource.onerror = () => {
  setTimeout(() => {
    // Reconnect
    startTeaching();
  }, 1000);
};
```

### Q2: 图片加载失败

**原因**: 图片ID不存在或URL过期

**解决**: 显示占位图和错误提示

```tsx
<img
  src={imageUrl}
  onError={(e) => {
    e.target.src = '/placeholder.png';
  }}
/>
```

### Q3: 事件顺序错乱

**原因**: 异步事件处理顺序问题

**解决**: 添加事件序号

```typescript
interface TeachingEvent {
  sequence: number;  // Add sequence number
  type: string;
  message?: string;
}

// Sort by sequence before rendering
const sortedEvents = events.sort((a, b) => a.sequence - b.sequence);
```
