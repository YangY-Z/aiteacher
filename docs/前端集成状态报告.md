// hooks/useTeachingImage.ts
export function useTeachingImage(imageId: string | null) {
  const [imageUrl, setImageUrl] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!imageId) return;
    
    setLoading(true);
    teachingV2Api.getImage(imageId)
      .then(data => setImageUrl(data.url))
      .catch(err => setError(err.message))
      .finally(() => setLoading(false));
  }, [imageId]);

  return { imageUrl, loading, error };
}

## 📝 具体实现建议

### 1. 创建自定义Hook

**文件**: `src/hooks/useTeachingV2.ts`

```typescript
export function useTeachingV2(sessionId: string) {
  const [useTools, setUseTools] = useState(true);
  const [events, setEvents] = useState<TeachingEvent[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const startTeaching = useCallback(async () => {
    setIsStreaming(true);
    setError(null);
    setEvents([]);

    try {
      if (useTools) {
        await teachingV2Api.startTeaching(sessionId, true, {
          onSegment: (data) => setEvents(prev => [...prev, data]),
          onComplete: () => setIsStreaming(false),
          onError: (err) => setError(err),
        });
      } else {
        await learningApi.streamTeachingContent(sessionId, {
          onSegment: (data) => setEvents(prev => [...prev, data]),
          onComplete: () => setIsStreaming(false),
          onError: (err) => setError(err),
        });
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
      setIsStreaming(false);
    }
  }, [sessionId, useTools]);

  return {
    events,
    isStreaming,
    error,
    useTools,
    setUseTools,
    startTeaching,
  };
}
```
