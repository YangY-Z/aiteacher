## 🎯 核心思想

### 问题
传统方案面临的两难困境：
- **预定义模板**：无法枚举所有教学场景
- **LLM直接生成代码**：质量不稳定、安全风险高

### 解决方案：分层抽象

```
确定性底层（原子元素）+ 不确定性顶层（LLM组合）= 灵活且安全
```

**关键突破**：
- ✅ 枚举**原子元素**（10个基本图形）
- ❌ 不枚举**所有图片**（无限种可能）
- ✅ LLM负责**组合**（无限可能）
- ✅ 底层保证**安全**（有限可控）

### 版本说明

| 版本 | 文件 | 特点 | 适用场景 |
|------|------|------|----------|
| **基础版** | `layered_abstraction_demo.py` | 简单线性布局，纯色填充 | 快速原型验证 |
| **增强版** | `enhanced_layered_demo.py` | 智能布局，渐变动画，视觉效果精美 | **生产环境部署** |

```python
class SmartComposer:
    """智能组合决策层"""
    
    def compose_process_layout(self, semantic_graph: Dict) -> Dict:
        """流程图布局：输入-处理-输出 三层结构"""
        
        entities = semantic_graph.get("entities", [])
        
        # 1. 自动分类实体
        inputs = []      # 能量源、物质
        processors = []  # 器官、细胞器
        outputs = []     # 产物
        
        for entity in entities:
            if entity.get("type") in ["能量源", "物质"]:
                inputs.append(entity)
            elif entity.get("type") == "产物":
                outputs.append(entity)
            elif entity.get("type") in ["器官", "细胞器"]:
                processors.append(entity)
        
        # 2. 三层布局
        # 输入区域（左侧上方）
        for i, entity in enumerate(inputs):
            x = 120
            y = 150 + i * 100
            self.add_entity_element(composition, entity, x, y, "input")
        
        # 处理区域（中心，放大）
        if processors:
            processor = processors[0]
            x, y = 500, 350  # 中心位置
            self.add_processor_element(composition, processor, x, y)
        
        # 输出区域（右侧下方）
        for i, entity in enumerate(outputs):
            x = 880
            y = 150 + i * 120
            self.add_entity_element(composition, entity, x, y, "output")
        
        # 3. 添加曲线连接
        self.add_connection_arrows(composition, inputs, processors, outputs)
        
        return composition
    
    def get_fill_for_entity(self, entity: Dict, category: str) -> str:
        """智能颜色选择"""
        name = entity.get("name", "")
        
        if name == "太阳":
            return "url(#sunGradient)"      # 太阳用金色渐变
        elif name == "叶片":
            return "url(#leafGradient)"     # 叶片用绿色渐变
        elif category == "output":
            if "O₂" in name:
                return "#4caf50"            # 氧气用绿色
            else:
                return "#e91e63"            # 葡萄糖用粉色
        elif category == "input":
            if "H₂O" in name:
                return "#2196f3"            # 水用蓝色
            else:
                return "#9e9e9e"            # CO2用灰色
        
        return "#2196f3"
```

## 💻 完整代码实现

### 第一层：原子渲染层（底层）

**职责**：只负责渲染基本图形，不包含任何业务逻辑

**特点**：
- ✅ 100%确定性（纯函数）
- ✅ 无副作用
- ✅ 完全可预测
- ✅ 100%安全

```python
from dataclasses import dataclass
from typing import Dict, List, Tuple, Optional
import json
import re

@dataclass
class ElementParams:
    """元素参数"""
    element_type: str
    params: Dict[str, any]


class AtomicRenderer:
    """
    原子渲染层：只渲染基本图形，不包含任何逻辑
    
    设计原则：
    1. 纯函数：相同输入总是产生相同输出
    2. 无副作用：不修改外部状态
    3. 类型安全：所有参数都有类型检查
    4. 范围限制：坐标、颜色等都在安全范围内
    """
    
    # 支持的原子元素类型（白名单）
    SUPPORTED_ELEMENTS = {
        "circle",      # 圆形
        "ellipse",     # 椭圆
        "rectangle",   # 矩形
        "line",        # 线段
        "arrow",       # 箭头
        "text",        # 文本
        "polygon",     # 多边形
        "path"         # 路径（受限）
    }
    
    # 支持的颜色格式（白名单）
    COLOR_PATTERN = re.compile(r'^#[0-9A-Fa-f]{6}$')
    
    # 坐标范围限制
    COORDINATE_RANGE = {
        "x": (0, 2000),
        "y": (0, 2000)
    }
    
    # 尺寸范围限制
    SIZE_RANGE = {
        "radius": (0.1, 1000),
        "width": (0.1, 2000),
        "height": (0.1, 2000)
    }
    
    def render(self, composition: Dict) -> str:
        """
        渲染组合方案
        
        Args:
            composition: 组合方案，包含元素列表
        
        Returns:
            SVG字符串
        """
        # 验证输入
        if not self.validate_composition(composition):
            raise ValueError("Invalid composition")
        
        # 生成SVG
        svg_parts = [
            '<svg viewBox="0 0 1000 600" xmlns="http://www.w3.org/2000/svg">',
            '<!-- Generated by AtomicRenderer -->'
        ]
        
        # 渲染每个元素
        for element in composition.get("elements", []):
            svg_element = self.render_element(element)
            svg_parts.append(svg_element)
        
        svg_parts.append('</svg>')
        
        return '\n'.join(svg_parts)
    
    def render_element(self, element: Dict) -> str:
        """
        渲染单个元素（路由到具体渲染函数）
        """
        element_type = element["type"]
        params = element["params"]
        
        # 类型检查（白名单）
        if element_type not in self.SUPPORTED_ELEMENTS:
            raise ValueError(f"Unsupported element type: {element_type}")
        
        # 渲染器路由
        renderers = {
            "circle": self.render_circle,
            "ellipse": self.render_ellipse,
            "rectangle": self.render_rectangle,
            "line": self.render_line,
            "arrow": self.render_arrow,
            "text": self.render_text,
            "polygon": self.render_polygon,
            "path": self.render_path
        }
        
        renderer = renderers.get(element_type)
        if not renderer:
            raise ValueError(f"No renderer for type: {element_type}")
        
        return renderer(params)
    
    # ========== 具体渲染函数 ==========
    
    def render_circle(self, params: Dict) -> str:
        """
        渲染圆形
        
        必需参数：
        - cx: 圆心x坐标
        - cy: 圆心y坐标
        - r: 半径
        
        可选参数：
        - fill: 填充颜色（默认#000）
        - stroke: 边框颜色
        - stroke_width: 边框宽度
        - opacity: 透明度
        """
        # 参数验证和范围限制
        cx = self.clamp_coordinate(params["cx"], "x")
        cy = self.clamp_coordinate(params["cy"], "y")
        r = self.clamp_size(params["r"], "radius")
        
        # 可选参数
        fill = self.validate_color(params.get("fill", "#000000"))
        stroke = self.validate_color(params.get("stroke", ""))
        stroke_width = params.get("stroke_width", 0)
        opacity = self.clamp(params.get("opacity", 1.0), 0, 1)
        
        # 生成SVG
        attrs = [
            f'cx="{cx}"',
            f'cy="{cy}"',
            f'r="{r}"',
            f'fill="{fill}"',
            f'opacity="{opacity}"'
        ]
        
        if stroke:
            attrs.append(f'stroke="{stroke}"')
            attrs.append(f'stroke-width="{stroke_width}"')
        
        return f'<circle {" ".join(attrs)}/>'
    
    def render_ellipse(self, params: Dict) -> str:
        """
        渲染椭圆
        
        必需参数：
        - cx: 圆心x坐标
        - cy: 圆心y坐标
        - rx: x轴半径
        - ry: y轴半径
        """
        cx = self.clamp_coordinate(params["cx"], "x")
        cy = self.clamp_coordinate(params["cy"], "y")
        rx = self.clamp_size(params["rx"], "radius")
        ry = self.clamp_size(params["ry"], "radius")
        
        fill = self.validate_color(params.get("fill", "#000000"))
        stroke = self.validate_color(params.get("stroke", ""))
        stroke_width = params.get("stroke_width", 0)
        opacity = self.clamp(params.get("opacity", 1.0), 0, 1)
        
        attrs = [
            f'cx="{cx}"',
            f'cy="{cy}"',
            f'rx="{rx}"',
            f'ry="{ry}"',
            f'fill="{fill}"',
            f'opacity="{opacity}"'
        ]
        
        if stroke:
            attrs.append(f'stroke="{stroke}"')
            attrs.append(f'stroke-width="{stroke_width}"')
        
        return f'<ellipse {" ".join(attrs)}/>'
    
    def render_rectangle(self, params: Dict) -> str:
        """
        渲染矩形
        
        必需参数：
        - x: 左上角x坐标
        - y: 左上角y坐标
        - width: 宽度
        - height: 高度
        
        可选参数：
        - rx: 圆角x半径
        - ry: 圆角y半径
        """
        x = self.clamp_coordinate(params["x"], "x")
        y = self.clamp_coordinate(params["y"], "y")
        width = self.clamp_size(params["width"], "width")
        height = self.clamp_size(params["height"], "height")
        
        fill = self.validate_color(params.get("fill", "#000000"))
        stroke = self.validate_color(params.get("stroke", ""))
        stroke_width = params.get("stroke_width", 0)
        opacity = self.clamp(params.get("opacity", 1.0), 0, 1)
        rx = params.get("rx", 0)
        ry = params.get("ry", 0)
        
        attrs = [
            f'x="{x}"',
            f'y="{y}"',
            f'width="{width}"',
            f'height="{height}"',
            f'fill="{fill}"',
            f'opacity="{opacity}"'
        ]
        
        if rx > 0:
            attrs.append(f'rx="{rx}"')
        if ry > 0:
            attrs.append(f'ry="{ry}"')
        if stroke:
            attrs.append(f'stroke="{stroke}"')
            attrs.append(f'stroke-width="{stroke_width}"')
        
        return f'<rect {" ".join(attrs)}/>'
    
    def render_line(self, params: Dict) -> str:
        """
        渲染线段
        
        必需参数：
        - x1, y1: 起点
        - x2, y2: 终点
        """
        x1 = self.clamp_coordinate(params["x1"], "x")
        y1 = self.clamp_coordinate(params["y1"], "y")
        x2 = self.clamp_coordinate(params["x2"], "x")
        y2 = self.clamp_coordinate(params["y2"], "y")
        
        stroke = self.validate_color(params.get("stroke", "#000000"))
        stroke_width = params.get("stroke_width", 1)
        stroke_dasharray = params.get("stroke_dasharray", "")
        
        attrs = [
            f'x1="{x1}"',
            f'y1="{y1}"',
            f'x2="{x2}"',
            f'y2="{y2}"',
            f'stroke="{stroke}"',
            f'stroke-width="{stroke_width}"'
        ]
        
        if stroke_dasharray:
            attrs.append(f'stroke-dasharray="{stroke_dasharray}"')
        
        return f'<line {" ".join(attrs)}/>'
    
    def render_arrow(self, params: Dict) -> str:
        """
        渲染箭头（线段 + 三角形）
        
        必需参数：
        - x1, y1: 起点
        - x2, y2: 终点
        """
        x1 = self.clamp_coordinate(params["x1"], "x")
        y1 = self.clamp_coordinate(params["y1"], "y")
        x2 = self.clamp_coordinate(params["x2"], "x")
        y2 = self.clamp_coordinate(params["y2"], "y")
        
        stroke = self.validate_color(params.get("stroke", "#000000"))
        stroke_width = params.get("stroke_width", 2)
        
        # 计算箭头头部（三角形）
        import math
        angle = math.atan2(y2 - y1, x2 - x1)
        arrow_length = 10
        arrow_angle = math.pi / 6  # 30度
        
        # 箭头两个点
        x3 = x2 - arrow_length * math.cos(angle - arrow_angle)
        y3 = y2 - arrow_length * math.sin(angle - arrow_angle)
        x4 = x2 - arrow_length * math.cos(angle + arrow_angle)
        y4 = y2 - arrow_length * math.sin(angle + arrow_angle)
        
        # 组合：线段 + 三角形
        svg = f'''
<line x1="{x1}" y1="{y1}" x2="{x2}" y2="{y2}" stroke="{stroke}" stroke-width="{stroke_width}"/>
<polygon points="{x2},{y2} {x3},{y3} {x4},{y4}" fill="{stroke}"/>
'''
        return svg.strip()
    
    def render_text(self, params: Dict) -> str:
        """
        渲染文本
        
        必需参数：
        - x, y: 坐标
        - content: 文本内容
        
        安全限制：
        - 内容长度：最大200字符
        - 禁止HTML标签
        - 禁止JavaScript代码
        """
        x = self.clamp_coordinate(params["x"], "x")
        y = self.clamp_coordinate(params["y"], "y")
        content = self.sanitize_text(params["content"])
        
        font_size = self.clamp(params.get("font_size", 16), 8, 72)
        fill = self.validate_color(params.get("fill", "#000000"))
        font_weight = params.get("font_weight", "normal")
        text_anchor = params.get("text_anchor", "start")
        
        attrs = [
            f'x="{x}"',
            f'y="{y}"',
            f'font-size="{font_size}"',
            f'fill="{fill}"',
            f'font-weight="{font_weight}"',
            f'text-anchor="{text_anchor}"'
        ]
        
        return f'<text {" ".join(attrs)}>{content}</text>'
    
    def render_polygon(self, params: Dict) -> str:
        """
        渲染多边形
        
        必需参数：
        - points: 顶点列表 [[x1,y1], [x2,y2], ...]
        """
        points = params["points"]
        
        # 验证并限制每个顶点坐标
        validated_points = []
        for point in points:
            if len(point) != 2:
                raise ValueError(f"Invalid point: {point}")
            x = self.clamp_coordinate(point[0], "x")
            y = self.clamp_coordinate(point[1], "y")
            validated_points.append(f"{x},{y}")
        
        points_str = " ".join(validated_points)
        
        fill = self.validate_color(params.get("fill", "#000000"))
        stroke = self.validate_color(params.get("stroke", ""))
        stroke_width = params.get("stroke_width", 0)
        opacity = self.clamp(params.get("opacity", 1.0), 0, 1)
        
        attrs = [
            f'points="{points_str}"',
            f'fill="{fill}"',
            f'opacity="{opacity}"'
        ]
        
        if stroke:
            attrs.append(f'stroke="{stroke}"')
            attrs.append(f'stroke-width="{stroke_width}"')
        
        return f'<polygon {" ".join(attrs)}/>'
    
    def render_path(self, params: Dict) -> str:
        """
        渲染路径（受限）
        
        安全限制：
        - 只允许 M, L, Q, C, A, Z 命令
        - 禁止相对坐标
        - 坐标范围限制
        """
        d = params["d"]
        
        # 验证路径命令（白名单）
        allowed_commands = {'M', 'L', 'Q', 'C', 'A', 'Z'}
        for cmd in re.findall(r'[A-Z]', d):
            if cmd not in allowed_commands:
                raise ValueError(f"Disallowed path command: {cmd}")
        
        # 禁止相对坐标
        if re.search(r'[a-z]', d):
            raise ValueError("Relative coordinates not allowed")
        
        fill = self.validate_color(params.get("fill", "none"))
        stroke = self.validate_color(params.get("stroke", "#000000"))
        stroke_width = params.get("stroke_width", 1)
        
        attrs = [
            f'd="{d}"',
            f'fill="{fill}"',
            f'stroke="{stroke}"',
            f'stroke-width="{stroke_width}"'
        ]
        
        return f'<path {" ".join(attrs)}/>'
    
    # ========== 辅助函数 ==========
    
    def validate_composition(self, composition: Dict) -> bool:
        """验证组合方案结构"""
        if "elements" not in composition:
            return False
        
        if not isinstance(composition["elements"], list):
            return False
        
        for element in composition["elements"]:
            if "type" not in element or "params" not in element:
                return False
        
        return True
    
    def validate_color(self, color: str) -> str:
        """
        验证颜色格式
        
        只允许十六进制颜色：#RRGGBB
        """
        if not color:
            return "#000000"
        
        if not self.COLOR_PATTERN.match(color):
            raise ValueError(f"Invalid color format: {color}")
        
        return color
    
    def clamp_coordinate(self, value: float, axis: str) -> float:
        """
        限制坐标范围
        """
        min_val, max_val = self.COORDINATE_RANGE[axis]
        return max(min_val, min(max_val, value))
    
    def clamp_size(self, value: float, size_type: str) -> float:
        """
        限制尺寸范围
        """
        min_val, max_val = self.SIZE_RANGE[size_type]
        return max(min_val, min(max_val, value))
    
    def clamp(self, value: float, min_val: float, max_val: float) -> float:
        """通用范围限制"""
        return max(min_val, min(max_val, value))
    
    def sanitize_text(self, text: str) -> str:
        """
        清理文本内容（安全过滤）
        """
        # 长度限制
        if len(text) > 200:
            raise ValueError("Text too long")
        
        # 禁止HTML标签
        if '<' in text or '>' in text:
            raise ValueError("HTML tags not allowed in text")
        
        # 禁止JavaScript
        if 'script' in text.lower():
            raise ValueError("JavaScript not allowed in text")
        
        # 转义特殊字符
        text = text.replace('&', '&amp;')
        text = text.replace('"', '&quot;')
        text = text.replace("'", '&apos;')
        
        return text
```

### 第二层：组合决策层（中层）

**职责**：根据语义图生成具体的渲染方案

**特点**：
- ✅ 半确定性（在约束范围内）
- ✅ LLM辅助决策
- ✅ 参数验证
- ✅ 自动修正

```python
import json
from typing import Dict, List

class ComponentComposer:
    """
    组合决策层：将语义图转为渲染方案
    
    设计原则：
    1. 约束驱动：在安全边界内生成方案
    2. 自动修正：超出约束的参数自动修正
    3. 策略选择：根据场景选择最佳组合策略
    4. 可验证性：所有方案都可验证
    """
    
    # 布局策略
    LAYOUT_STRATEGIES = {
        "flow": "从左到右流程图",
        "tree": "树状层次结构",
        "radial": "放射状分布",
        "grid": "网格布局"
    }
    
    # 配色方案（预定义，安全）
    COLOR_SCHEMES = {
        "natural": {
            "primary": "#4caf50",
            "secondary": "#81c784",
            "accent": "#ff9800"
        },
        "tech": {
            "primary": "#2196f3",
            "secondary": "#64b5f6",
            "accent": "#ff5722"
        },
        "minimal": {
            "primary": "#333333",
            "secondary": "#666666",
            "accent": "#000000"
        }
    }
    
    def __init__(self, llm_client=None):
        self.llm_client = llm_client
        self.atomic_renderer = AtomicRenderer()
    
    async def compose(self, semantic_graph: Dict, constraints: Dict = None) -> Dict:
        """
        将语义图转为渲染方案
        
        Args:
            semantic_graph: 语义图（来自顶层）
            constraints: 额外约束（可选）
        
        Returns:
            组合方案（传给底层渲染）
        """
        # 1. 分析语义图
        analysis = self.analyze_semantic_graph(semantic_graph)
        
        # 2. 选择布局策略
        layout_strategy = self.select_layout_strategy(analysis)
        
        # 3. LLM生成初始方案（如果可用）
        if self.llm_client:
            initial_composition = await self.generate_with_llm(
                semantic_graph, 
                layout_strategy
            )
        else:
            # 无LLM时使用规则引擎
            initial_composition = self.generate_with_rules(
                semantic_graph,
                layout_strategy
            )
        
        # 4. 验证和修正
        validated_composition = self.validate_and_correct(initial_composition)
        
        return validated_composition
    
    def analyze_semantic_graph(self, semantic_graph: Dict) -> Dict:
        """
        分析语义图特征
        """
        analysis = {
            "type": semantic_graph.get("type", "unknown"),
            "entity_count": len(semantic_graph.get("entities", [])),
            "relation_count": len(semantic_graph.get("relations", [])),
            "process_count": len(semantic_graph.get("process", [])),
            "complexity": "low"
        }
        
        # 判断复杂度
        total_elements = analysis["entity_count"] + analysis["relation_count"]
        if total_elements > 20:
            analysis["complexity"] = "high"
        elif total_elements > 10:
            analysis["complexity"] = "medium"
        
        return analysis
    
    def select_layout_strategy(self, analysis: Dict) -> str:
        """
        根据分析结果选择布局策略
        """
        # 规则驱动
        if analysis["type"] == "process":
            return "flow"
        elif analysis["type"] == "structure":
            return "tree"
        elif analysis["type"] == "relationship":
            if analysis["complexity"] == "high":
                return "radial"
            else:
                return "tree"
        else:
            return "grid"
    
    async def generate_with_llm(self, semantic_graph: Dict, layout_strategy: str) -> Dict:
        """
        使用LLM生成组合方案
        """
        prompt = f"""
        语义图：{json.dumps(semantic_graph, ensure_ascii=False)}
        布局策略：{layout_strategy}
        
        请生成SVG渲染方案，使用以下原子元素：
        
        可用元素类型：
        - circle: 圆形
        - ellipse: 椭圆
        - rectangle: 矩形
        - line: 线段
        - arrow: 箭头
        - text: 文本
        - polygon: 多边形
        
        约束：
        - 坐标范围：x(0-1000), y(0-600)
        - 颜色格式：#RRGGBB
        - 文本长度：最大200字符
        - 禁止HTML标签和JavaScript
        
        输出JSON格式：
        {{
            "elements": [
                {{
                    "type": "circle",
                    "params": {{
                        "cx": 100,
                        "cy": 100,
                        "r": 50,
                        "fill": "#ffeb3b"
                    }}
                }},
                {{
                    "type": "text",
                    "params": {{
                        "x": 100,
                        "y": 110,
                        "content": "太阳",
                        "font_size": 16
                    }}
                }}
            ]
        }}
        """
        
        composition_json = await self.llm_client.generate_json(prompt)
        
        return composition_json
    
    def generate_with_rules(self, semantic_graph: Dict, layout_strategy: str) -> Dict:
        """
        使用规则引擎生成组合方案（无LLM降级方案）
        """
        composition = {"elements": []}
        
        entities = semantic_graph.get("entities", [])
        
        if layout_strategy == "flow":
            # 从左到右布局
            for i, entity in enumerate(entities):
                x = 100 + i * 150
                y = 300
                
                # 圆形表示实体
                composition["elements"].append({
                    "type": "circle",
                    "params": {
                        "cx": x,
                        "cy": y,
                        "r": 40,
                        "fill": "#4caf50"
                    }
                })
                
                # 文本标签
                composition["elements"].append({
                    "type": "text",
                    "params": {
                        "x": x,
                        "y": y + 60,
                        "content": entity["name"],
                        "font_size": 14,
                        "text_anchor": "middle"
                    }
                })
                
                # 箭头连接（除了最后一个）
                if i < len(entities) - 1:
                    composition["elements"].append({
                        "type": "arrow",
                        "params": {
                            "x1": x + 40,
                            "y1": y,
                            "x2": x + 110,
                            "y2": y,
                            "stroke": "#666666"
                        }
                    })
        
        # 其他布局策略...
        
        return composition
    
    def validate_and_correct(self, composition: Dict) -> Dict:
        """
        验证和自动修正组合方案
        """
        if "elements" not in composition:
            composition["elements"] = []
        
        validated_elements = []
        
        for element in composition.get("elements", []):
            try:
                # 验证类型
                if element["type"] not in self.atomic_renderer.SUPPORTED_ELEMENTS:
                    continue  # 跳过不支持的类型
                
                # 验证参数
                validated_params = self.validate_params(
                    element["type"],
                    element["params"]
                )
                
                validated_elements.append({
                    "type": element["type"],
                    "params": validated_params
                })
            
            except Exception as e:
                # 记录错误，跳过该元素
                print(f"Validation error: {e}")
                continue
        
        composition["elements"] = validated_elements
        
        return composition
    
    def validate_params(self, element_type: str, params: Dict) -> Dict:
        """
        验证和修正参数
        """
        validated = {}
        
        for key, value in params.items():
            # 坐标验证
            if key in ["cx", "cy", "x", "y", "x1", "y1", "x2", "y2"]:
                validated[key] = self.atomic_renderer.clamp_coordinate(
                    float(value),
                    "x" if "x" in key else "y"
                )
            
            # 尺寸验证
            elif key in ["r", "rx", "ry"]:
                validated[key] = self.atomic_renderer.clamp_size(
                    float(value),
                    "radius"
                )
            elif key in ["width", "height"]:
                validated[key] = self.atomic_renderer.clamp_size(
                    float(value),
                    key
                )
            
            # 颜色验证
            elif key in ["fill", "stroke"]:
                validated[key] = self.atomic_renderer.validate_color(value)
            
            # 文本验证
            elif key == "content":
                validated[key] = self.atomic_renderer.sanitize_text(value)
            
            # 其他参数（直接传递）
            else:
                validated[key] = value
        
        return validated
```

### 第三层：语义理解层（顶层）

**职责**：理解自然语言，提取语义结构

**特点**：
- ✅ LLM驱动（不确定性高）
- ✅ 语义提取（不涉及渲染细节）
- ✅ 抽象描述（平台无关）

```python
from typing import Dict, List

class SemanticAnalyzer:
    """
    语义理解层：理解自然语言，提取语义结构
    
    设计原则：
    1. 语义优先：只关注"要表达什么"，不关注"怎么表达"
    2. 平台无关：输出与具体渲染技术无关
    3. 结构化：输出结构化语义图
    """
    
    def __init__(self, llm_client):
        self.llm_client = llm_client
    
    async def analyze(self, natural_language: str) -> Dict:
        """
        分析自然语言，生成语义图
        
        Args:
            natural_language: 自然语言描述
        
        Returns:
            语义图（传给中层）
        """
        prompt = f"""
        分析以下描述，提取语义结构：
        
        "{natural_language}"
        
        输出JSON格式的语义图：
        {{
            "type": "process" | "structure" | "relationship" | "comparison",
            "entities": [
                {{
                    "name": "实体名称",
                    "type": "实体类型（如：物质、器官、概念）",
                    "attributes": ["属性1", "属性2"],
                    "importance": 1-5  # 重要程度
                }}
            ],
            "relations": [
                {{
                    "from": "实体1名称",
                    "to": "实体2名称",
                    "relation": "关系类型（如：输入、输出、包含）",
                    "label": "关系标签（可选）"
                }}
            ],
            "process": [
                {{
                    "step": 步骤编号,
                    "action": "动作描述",
                    "objects": ["涉及的对象"],
                    "result": "结果"
                }}
            ],
            "annotations": [
                {{
                    "text": "说明文字",
                    "target": "指向的实体或关系",
                    "position": "top" | "bottom" | "left" | "right"
                }}
            ]
        }}
        
        注意：
        1. 只描述"要展示什么"，不描述"怎么展示"
        2. 不要包含坐标、颜色、大小等渲染细节
        3. 语义图应该与具体的可视化技术无关
        """
        
        semantic_graph = await self.llm_client.generate_json(prompt)
        
        # 后处理：验证和增强
        semantic_graph = self.post_process(semantic_graph)
        
        return semantic_graph
    
    def post_process(self, semantic_graph: Dict) -> Dict:
        """
        后处理：验证和增强语义图
        """
        # 确保必需字段存在
        if "type" not in semantic_graph:
            semantic_graph["type"] = "unknown"
        
        if "entities" not in semantic_graph:
            semantic_graph["entities"] = []
        
        if "relations" not in semantic_graph:
            semantic_graph["relations"] = []
        
        if "process" not in semantic_graph:
            semantic_graph["process"] = []
        
        # 验证实体完整性
        validated_entities = []
        for entity in semantic_graph["entities"]:
            if "name" not in entity:
                continue
            
            # 填充默认值
            entity.setdefault("type", "unknown")
            entity.setdefault("attributes", [])
            entity.setdefault("importance", 3)
            
            validated_entities.append(entity)
        
        semantic_graph["entities"] = validated_entities
        
        # 验证关系完整性
        validated_relations = []
        entity_names = {e["name"] for e in validated_entities}
        
        for relation in semantic_graph["relations"]:
            if "from" not in relation or "to" not in relation:
                continue
            
            # 检查实体是否存在
            if relation["from"] not in entity_names or relation["to"] not in entity_names:
                continue
            
            relation.setdefault("relation", "related")
            relation.setdefault("label", "")
            
            validated_relations.append(relation)
        
        semantic_graph["relations"] = validated_relations
        
        return semantic_graph
```

### 4. 性能优化

```python
# 缓存机制
class CachedLayeredSystem:
    """
    带缓存的分层系统
    """
    def __init__(self):
        self.semantic_cache = {}  # 语义图缓存
        self.composition_cache = {}  # 组合方案缓存
        self.renderer = AtomicRenderer()
    
    async def visualize(self, user_input: str):
        # 1. 检查语义缓存
        if user_input in self.semantic_cache:
            semantic_graph = self.semantic_cache[user_input]
        else:
            semantic_graph = await self.semantic_analyzer.analyze(user_input)
            self.semantic_cache[user_input] = semantic_graph
        
        # 2. 检查组合缓存
        cache_key = hash(json.dumps(semantic_graph))
        if cache_key in self.composition_cache:
            composition = self.composition_cache[cache_key]
        else:
            composition = await self.composer.compose(semantic_graph)
            self.composition_cache[cache_key] = composition
        
        # 3. 渲染（底层总是快速）
        svg = self.renderer.render(composition)
        
        return svg
```
